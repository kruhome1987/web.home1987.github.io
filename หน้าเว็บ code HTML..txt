    // ดึงข้อมูลลิงก์
    function fetchLinks(manualClick = false) {
      updateConnectionStatus("loading");
      
      if (manualClick) {
        Swal.fire({
          title: 'กำลังดึงข้อมูล...',
          text: 'กรุณารอสักครู่ กำลังเรียกข้อมูลจาก Google Sheets',
          allowOutsideClick: false,
          didOpen: () => {
            Swal.showLoading();
          }
        });
      }

      // ตรวจสอบข้อมูลแบบ GET จาก Apps Script
      fetch(APPS_SCRIPT_URL)
        .then(response => {
          if (!response.ok) {
            throw new Error("Network response was not ok");
          }
          return response.json();
        })
        .then(res => {
          if (res.status === "success") {
            links = res.data;
            updateConnectionStatus("online");
            localStorage.setItem('webdir_links_cache', JSON.stringify(links));
            localStorage.setItem('webdir_is_mock', 'false');
            
            if (manualClick) {
              Swal.fire({
                icon: 'success',
                title: 'สำเร็จ!',
                text: 'ดึงข้อมูลจาก Google Sheet สำเร็จแล้ว',
                timer: 1500,
                showConfirmButton: false
              });
            }
            renderApp();
          } else {
            throw new Error(res.message || "Unknown error from Apps Script");
          }
        })
        .catch(err => {
          console.warn("Apps Script API error, fallback to cache or mock data:", err);
          
          // ตรวจสอบว่ามี Cache เก่าเก็บไว้หรือไม่
          const cache = localStorage.getItem('webdir_links_cache');
          const isMockStorage = localStorage.getItem('webdir_is_mock') === 'true';
          
          if (cache && !isMockStorage) {
            links = JSON.parse(cache);
            updateConnectionStatus("online"); // หรือแสดงว่าใช้ออฟไลน์ cache
            if (manualClick) {
              Swal.fire({
                icon: 'warning',
                title: 'ใช้ข้อมูลสำรอง',
                text: 'ไม่สามารถเชื่อมต่อ Sheet ได้ในขณะนี้ ระบบดึงข้อมูลล่าสุดที่บันทึกไว้ในอุปกรณ์ขึ้นมาแสดง',
              });
            }
          } else {

                icon: 'info',
                title: 'โหลดข้อมูลจำลองสำเร็จ',

              });
            }
          }
          renderApp();
        });
    }

    // สุ่มสีพื้นหลัง gradient สำหรับการ์ดที่ไม่มีรูปภาพประกอบ
    function generateGradientFromString(str) {
      let hash = 0;
      for (let i = 0; i < str.length; i++) {
        hash = str.charCodeAt(i) + ((hash << 5) - hash);
      }
      const colors = [
        ['from-blue-500', 'to-cyan-500'],
        ['from-purple-500', 'to-indigo-500'],
        ['from-pink-500', 'to-rose-500'],
        ['from-orange-500', 'to-amber-500'],
        ['from-emerald-500', 'to-teal-500'],
        ['from-indigo-500', 'to-purple-500']
      ];
      const index = Math.abs(hash) % colors.length;
      return colors[index];
    }

    // ฟังก์ชันแปลงลิงก์ Google Drive ให้เป็น direct link สำหรับแสดงในแท็ก img
    function convertDriveUrl(url) {
      if (!url) return url;
      
      // รูปแบบที่ 1: drive.google.com/file/d/FILE_ID/view...
      const fileDMatch = url.match(/\/file\/d\/([a-zA-Z0-9_-]+)/);
      if (fileDMatch && fileDMatch[1]) {
        return `https://lh3.googleusercontent.com/d/${fileDMatch[1]}`;
      }
      
      // รูปแบบที่ 2: drive.google.com/open?id=FILE_ID
      const openIdMatch = url.match(/[\?&]id=([a-zA-Z0-9_-]+)/);
      if (url.includes('drive.google.com') && openIdMatch && openIdMatch[1]) {
        return `https://lh3.googleusercontent.com/d/${openIdMatch[1]}`;
      }
      
      // รูปแบบที่ 3: docs.google.com/file/d/FILE_ID...
      const docsDMatch = url.match(/\/d\/([a-zA-Z0-9_-]+)/);
      if (url.includes('docs.google.com') && docsDMatch && docsDMatch[1]) {
        return `https://lh3.googleusercontent.com/d/${docsDMatch[1]}`;
      }

      return url;
    }

    // ฟังก์ชันสร้างหน้าตา gradient กรณีไม่ใส่รูปภาพ
    function getFallbackGradientHTML(title) {
      const gradient = generateGradientFromString(title);
      const firstLetter = title.charAt(0).toUpperCase();
      return `
        <div class="w-full h-full bg-gradient-to-br ${gradient[0]} ${gradient[1]} flex flex-col items-center justify-center text-white">
          <span class="text-4xl font-extrabold tracking-widest drop-shadow">${firstLetter}</span>
        </div>
      `;
    }

    // วาดการแบ่งหน้า
    function renderPagination() {
      const paginationContainer = document.getElementById('pagination');
      paginationContainer.innerHTML = "";

      const totalPages = Math.ceil(links.length / itemsPerPage);
      if (totalPages <= 1) {
        paginationContainer.parentElement.classList.add('hidden');
        return;
      }
      paginationContainer.parentElement.classList.remove('hidden');

      // ปุ่มย้อนกลับ (Previous)
      const prevBtn = document.createElement('button');
      prevBtn.disabled = currentPage === 1;
      prevBtn.className = `h-9 w-9 rounded-full flex items-center justify-center text-sm transition-all ${
        currentPage === 1 
          ? 'text-slate-300 dark:text-slate-600 cursor-not-allowed' 
          : 'text-slate-600 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700'
      }`;
      prevBtn.innerHTML = `<i class="fa-solid fa-chevron-left text-xs"></i>`;
      prevBtn.onclick = () => {
        if (currentPage > 1) {
          currentPage--;
          renderApp();
          window.scrollTo({ top: 300, behavior: 'smooth' });
        }
      };
      paginationContainer.appendChild(prevBtn);

      // สร้างเลขหน้า
      for (let i = 1; i <= totalPages; i++) {
        const pageBtn = document.createElement('button');
        pageBtn.className = `h-9 min-w-[36px] px-2.5 rounded-full text-xs font-semibold transition-all ${
          currentPage === i
            ? 'bg-gradient-to-r from-indigo-500 to-purple-600 text-white shadow-md'
            : 'text-slate-600 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700'
        }`;
        pageBtn.innerText = i;
        pageBtn.onclick = () => {
          currentPage = i;
          renderApp();
          window.scrollTo({ top: 300, behavior: 'smooth' });
        };
        paginationContainer.appendChild(pageBtn);
      }

      // ปุ่มถัดไป (Next)
      const nextBtn = document.createElement('button');
      nextBtn.disabled = currentPage === totalPages;
      nextBtn.className = `h-9 w-9 rounded-full flex items-center justify-center text-sm transition-all ${
        currentPage === totalPages 
          ? 'text-slate-300 dark:text-slate-600 cursor-not-allowed' 
          : 'text-slate-600 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700'
      }`;
      nextBtn.innerHTML = `<i class="fa-solid fa-chevron-right text-xs"></i>`;
      nextBtn.onclick = () => {
        if (currentPage < totalPages) {
          currentPage++;
          renderApp();
          window.scrollTo({ top: 300, behavior: 'smooth' });
        }
      };
      paginationContainer.appendChild(nextBtn);
    }

          imageUrl: data.imageUrl,

    // ฟังก์ชันลบลิงก์จากฐานข้อมูล
    function deleteLink(id) {
      if (connectionStatus === "mock") {
        // ลบออฟไลน์
        links = links.filter(item => item.id !== id);
        saveMockLinksToStorage();
        // ปรับหน้าปัจจุบันกรณีข้อมูลหมดหน้าพอดี
        const maxPage = Math.ceil(links.length / itemsPerPage);
        if (currentPage > maxPage && maxPage > 0) {
          currentPage = maxPage;
        }
        renderApp();

        Swal.fire({
          icon: 'success',
          title: 'ลบข้อมูลสำเร็จ',
          text: 'ลบข้อมูลแบบจำลองเรียบร้อยแล้ว',
          timer: 1500,
          showConfirmButton: false
        });
      } else {
        // ส่งยิงขอลบจริงจาก Google Apps Script
        Swal.fire({
          title: 'กำลังลบข้อมูล...',
          text: 'กรุณารอสักครู่ ระบบกำลังสื่อสารกับ Google Sheet',
          allowOutsideClick: false,
          didOpen: () => {
            Swal.showLoading();
          }
        });

        fetch(APPS_SCRIPT_URL, {
          method: 'POST',
          mode: 'no-cors',
          headers: {
            'Content-Type': 'text/plain'
          },
          body: JSON.stringify({
            action: 'delete',
            password: adminPassword,
            id: id
          })
        })
        .then(() => {

            fetchLinks(false);
            Swal.fire({
              icon: 'success',
              title: 'ลบสำเร็จ',
              text: 'ลบข้อมูลจาก Google Sheet เรียบร้อยแล้ว',

              showConfirmButton: false
            });
          }, 1500);
        })

            icon: 'error',
